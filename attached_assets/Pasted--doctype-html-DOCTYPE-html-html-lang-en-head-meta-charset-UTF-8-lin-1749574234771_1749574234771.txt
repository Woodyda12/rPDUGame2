import * as THREE from 'https://unpkg.com/three@0.160.0/build/three.module.js';
const c=document.getElementById('c');
const scene=new THREE.Scene();
const camera=new THREE.PerspectiveCamera(70,800/600,0.1,100);camera.position.set(0,1.6,5);
const renderer=new THREE.WebGLRenderer({canvas:c});renderer.setSize(800,600);
scene.add(new THREE.AmbientLight(0xffffff));
const floor=new THREE.Mesh(new THREE.PlaneGeometry(10,10),new THREE.MeshStandardMaterial({color:0x444444}));floor.rotation.x=-Math.PI/2;scene.add(floor);
const keypad=new THREE.Mesh(new THREE.BoxGeometry(.5,.5,.2),new THREE.MeshStandardMaterial({color:0x888888}));keypad.position.set(-2,1,-1);scene.add(keypad);
const colors=['red','green','blue','yellow'].map((c,i)=>{const m=new THREE.Mesh(new THREE.BoxGeometry(.4,.4,.4),new THREE.MeshStandardMaterial({color:c}));m.position.set(-1.5+i*0.6,.3,0);scene.add(m);return m;});
const painting=new THREE.Mesh(new THREE.PlaneGeometry(1,1),new THREE.MeshStandardMaterial({color:0x999999}));painting.position.set(2,1,-2);painting.rotation.y=Math.PI/4;scene.add(painting);
const posts=[new THREE.Mesh(new THREE.CylinderGeometry(.1,.1,.5,16),new THREE.MeshStandardMaterial({color:0x333333})),new THREE.Mesh(new THREE.CylinderGeometry(.1,.1,.5,16),new THREE.MeshStandardMaterial({color:0x333333}))];posts[0].position.set(-1,.25,2);posts[1].position.set(1,.25,2);posts.forEach(p=>scene.add(p));
const ray=new THREE.Raycaster();const mouse=new THREE.Vector2();
let seq=[],solved=[0,0,0,0,0,0],start=Date.now();
function done(i){if(!solved[i]){solved[i]=1;if(solved.every(Boolean))alert(`Escaped in ${((Date.now()-start)/1000)|0} s – screenshot this!`);}}
function click(e){const r=c.getBoundingClientRect();mouse.x=((e.clientX-r.left)/r.width)*2-1;mouse.y=-((e.clientY-r.top)/r.height)*2+1;ray.setFromCamera(mouse,camera);const hit=ray.intersectObjects([keypad,painting,...colors]);if(!hit.length)return;const o=hit[0].object;if(o===keypad)done(prompt('Code?')==='314'?0:9);else if(o===painting)done(2);else if(colors.includes(o)){seq.push(o.material.color.getStyle()[0]);if(seq.join('').startsWith('rgb y'.replace(/ /g,''))&&seq.length===4)done(1);}}
c.addEventListener('pointerdown',click);
function animate(){requestAnimationFrame(animate);renderer.render(scene,camera);}animate();
document.getElementById('slider').oninput=e=>Math.abs(e.target.value-42)<=1&&done(3);
['g1','g2'].forEach(id=>document.getElementById(id).addEventListener('dragstart',ev=>ev.dataTransfer.setData('id',id)));
['p1','p2'].forEach(id=>{const el=document.getElementById(id);el.addEventListener('dragover',e=>e.preventDefault());el.addEventListener('drop',e=>{e.preventDefault();el.appendChild(document.getElementById(e.dataTransfer.getData('id')));if([...document.querySelectorAll('.drop')].every(d=>d.children.length))done(4);});});
document.getElementById('rotinput').addEventListener('input',e=>e.target.value.trim()==='748'&&done(5));
</script>
